function getMaxID(countMaxIDAddresses)
  local maxID      = nil
  local lastID     = 0
  local countMaxID = 0

  while true do
    local IDpointer = readPointer(countMaxIDAddresses + 8 * countMaxID)
    local IDvalue   = readInteger(IDpointer)

    if not IDvalue or IDvalue <= 0 then
      break
    end

    if IDvalue ~= lastID + 1 then
      break
    end

    maxID      = IDvalue
    lastID     = IDvalue
    countMaxID = countMaxID + 1
  end

  if maxID == nil then
    return 1
  else
    return maxID
  end
end
